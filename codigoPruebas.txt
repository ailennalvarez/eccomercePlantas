let mostrarModal = null;
const verModal = () => {
  //Para que no se repita la modal
  if (mostrarModal !== null){
    mostrarModal.remove();
  }

  mostrarModal = document.createElement('div');
  const div1 = document.createElement('div');
  div1.classList.add('modal');
  div1.tabIndex = -1;
  //
  const div2 = document.createElement('div');
  div2.classList.add('modal-dialog');
  //
  const div3 = document.createElement('div');
  div3.classList.add('modal-content');
  //
  const div4 = document.createElement("div");
  div4.classList.add('modal-header');
  const h1 = document.createElement('h1');
  h1.classList.add('modal-title');
  h1.innerText=`CARRITO DE COMPRAS`; 
  const button = document.createElement('button');
  button.type=("button");
  button.classList.add('btn-close');
  // 
  div1.append(div2);
   div1.append(div3);
   div3.append(div4);
   div4.append(h1);
   div4.append(button);
   
 
  //

  // el contenido 
       carrito.forEach((prod)=>{
        const modalBody = document.createElement('div');
        modalBody.classList.add('modal-body');
        const titulo = document.createElement('h5');
        titulo.textContent = `${prod.nombre}`
        //categoria
        const cat = document.createElement('p');
        cat.innerText=`${prod.categoria}`;
        //precio
        const precio = document.createElement('p');
        precio.classList.add('precio');
        precio.textContent=`$${prod.precio}`;
        //imagen
        const img = document.createElement('img');
        img.innerText=`${prod.imagen}`
      
        modalBody.append(titulo);
        modalBody.append(cat);
        modalBody.append(precio);
        modalBody.append(img);
        mostrarModal.appendChild(modalBody);
   })
   let total = carrito.reduce((acc,items)=>acc + items.precio,0)
   //footer
   const footer = document.createElement('div');
   footer.classList.add('modal-footer');
   const btnCerrar = document.createElement('button');
   btnCerrar.classList.add('btn','btn-secondary');
   const totalFinal = document.createElement('h3');
   totalFinal.textContent=`TOTAL $ ${total}`

   

   mostrarModal.append(footer);
   footer.append(button);
   footer.append(totalFinal);
   mostrarModal.appendChild(div1);
   modalContenedor.appendChild(mostrarModal)
 
  
   
}
verCarrito.addEventListener('click', verModal);
